## Benchmark

<DocsTextBlock className="m-b-medium-60">
  Benchmarked on 2025-11-02 across repeated runs of 100 icons. Results are
  averaged and rounded for readability.
</DocsTextBlock>

<DocsTextBlock className="m-b-medium-60">
  If you want to reproduce the analysis, run the following scripts:
</DocsTextBlock>

<DocsBanner
  title="Output directory"
  description="Make sure you have the /svg folder with your svg icon files in it at the root of your project before you run script."
  mode="warning"
/>

<DocsCodePreview
  language="bash"
  code={`echo "date,time,duration_s,components,memory_kb,bundle_kb" > benchmark-history.csv`}
/>

<DocsCodePreview
  language="bash"
  code={`for i in {1..100}; do
  echo "Run $i:"
    rm -rf ./icons

    START=$(date +%s%N)
    pnpx @usefui/svgjsx generate > /tmp/benchmark.log
    END=$(date +%s%N)

    DURATION=$(echo "scale=3; ($END - $START) / 1000000000" | bc)
    COMPONENTS=$(grep "Generated" /tmp/benchmark.log | tail -1 | grep -o '[0-9]* components' | grep -o '[0-9]*')
    MEMORY=$(grep "Memory used" /tmp/benchmark.log | grep -o '[0-9]* KB')
    BUNDLE_SIZE=$(du -sk icons/ | awk '{print $1}')

    echo "Execution Time: {DURATION}s"
    echo "Components: $COMPONENTS"
    echo "Memory: $MEMORY"
    echo "Bundle Size: {BUNDLE_SIZE}K"

    echo "$(date +%Y-%m-%d,%H:%M:%S),$DURATION,$COMPONENTS,$MEMORY,$BUNDLE_SIZE" >> benchmark-history.csv
    echo "Saved to benchmark-history.csv"

    echo "---"

done
`}
/>

<DocsCodePreview
  language="bash"
  code={`
  tail -n 100 benchmark-history.csv | awk -F',' '
  NR>1 {
    duration += $3
    memory += substr($4, 1, length($4)-3)
    count++
  }
  END {
    avg_duration = duration / count
    avg_memory = memory / count
    print "Average Duration:", avg_duration "s"
    print "Average Memory:", avg_memory " KB"
    print "Throughput:", 100 / avg_duration " icons/sec"
  }
`}
/>
