                <Badge
                    key={key}
                    variant={
                    variant as
                        | "primary"
                        | "secondary"
                        | "border"
                        | "error"
                        | "warning"
                        | "success"
                        | "meta"
                    }
                >
                    @fui
                </Badge>
                ))}
            </div>
            <div className="flex align-center g-medium-10 justify-center">
                {["square", "smooth", "round"].map((shape, key) => (
                <Badge key={key} shape={shape as "square" | "smooth" | "round"}>
                    @fui
                </Badge>
                ))}
            </div>
        </div>
    );

}

export default BadgeTemplate;
`} />
