                            | "mono"
                            | "border"

                        }
                    >
                    @fui
                    </Button>
                ),
                )}
            </div>
            <div className="flex align-center g-medium-10 justify-center">
                {["small", "medium", "large"].map((size, key) => (
                    <Button key={key} sizing={size as "small" | "medium" | "large"}>
                        @fui
                    </Button>
                ))}
            </div>
        </div>
    );

}

export default ButtonTemplate;

### Anatomy

<DocsTextBlock className="m-b-medium-60">
  The Button component **_is not_** built using the [Compound
  pattern](https://www.patterns.dev/react/compound-pattern/), it carries all the
  properties required to make to component work as intended.
</DocsTextBlock>
